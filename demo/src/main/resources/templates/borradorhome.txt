 class CategoryCarousel {
    constructor() {
        this.track = document.getElementById('carouselTrack');
        this.cards = document.querySelectorAll('.category-carousel-card');
        this.prevBtn = document.querySelector('.category-carousel-prev');
        this.nextBtn = document.querySelector('.category-carousel-next');
        
        this.cardWidth = 220; // Ancho de cada tarjeta + gap
        this.currentPosition = 0;
        this.visibleCards = this.calculateVisibleCards();
        this.maxPosition = (this.cards.length - this.visibleCards) * this.cardWidth;
        
        // Configurar eventos
        this.prevBtn.addEventListener('click', () => this.move(-1));
        this.nextBtn.addEventListener('click', () => this.move(1));
        window.addEventListener('resize', () => this.handleResize());
    }
    
    calculateVisibleCards() {
        const containerWidth = this.track.parentElement.offsetWidth - 40;
        return Math.floor(containerWidth / this.cardWidth);
    }
    
    move(direction) {
        this.currentPosition += direction * this.cardWidth * 2; // Mover 2 tarjetas a la vez
        
        // Ajustar l√≠mites
        if (this.currentPosition < 0) {
            this.currentPosition = 0;
        } else if (this.currentPosition > this.maxPosition) {
            this.currentPosition = this.maxPosition;
        }
        
        this.track.style.transform = `translateX(-${this.currentPosition}px)`;
        this.track.style.transition = 'transform 0.5s ease';
    }
    
    handleResize() {
        this.visibleCards = this.calculateVisibleCards();
        this.maxPosition = (this.cards.length - this.visibleCards) * this.cardWidth;
        
        if (this.currentPosition > this.maxPosition) {
            this.currentPosition = this.maxPosition;
            this.track.style.transform = `translateX(-${this.currentPosition}px)`;
        }
    }
}

            actualizarPrecioModal();
        });

        function actualizarPrecioModal() {
            if (!productoSeleccionado) return;
            let precioBase = parseFloat(productoSeleccionado.precio.toString().replace('$',''));
            let precioFinal = agrandarCombo ? (precioBase + 2.5) : precioBase;
            document.getElementById('modalProductoPrecio').innerText = '$' + precioFinal.toFixed(2);
        }

        // Funci√≥n para agregar al carrito (muestra modal con info)
function agregarAlCarrito(idProducto) {
    fetch('/api/usuario-logeado')
    .then(res => res.json())
    .then(data => {
        if (data.logeado) {
            const card = document.querySelector(`button[onclick*="agregarAlCarrito(${idProducto})"]`).closest('.card');
            const nombre = card.querySelector('.card-title').innerText;
            const descripcion = card.querySelector('.card-text').innerText;
            const precio = card.querySelector('.price-tag').innerText.replace('$','');
            const imagen = card.querySelector('.card-img-top').getAttribute('src');
            productoSeleccionado = { idProducto, nombre, descripcion, precio, imagen };

            // Reset cantidad y agrandar
            cantidadPedido = 1;
            agrandarCombo = false;
            document.getElementById('cantidadPedido').innerText = cantidadPedido;
            document.getElementById('toggleAgrandarCombo').checked = false;

            // Rellenar el modal
            document.getElementById('modalProductoImagen').src = imagen;
            document.getElementById('modalProductoNombre').innerText = nombre;
            document.getElementById('modalProductoDescripcion').innerText = descripcion;
            actualizarPrecioModal();

            // Cargar extras (productos con id_categoria 8 o 9) - CAMBIO AQU√ç
            const productosAdicionalesContainer = document.getElementById('productosAdicionalesContainer');
            productosAdicionalesContainer.innerHTML = '';
            const productos = window.productos;
            const productosAdicionales = productos.filter(p => p.idCategoria == 8 || p.idCategoria == 9);
            
            if (productosAdicionales.length > 0) {
                productosAdicionales.forEach(producto => {
                    const productoDiv = document.createElement('div');
                    productoDiv.className = "d-flex align-items-center justify-content-between mb-1";
                    productoDiv.innerHTML = `
                        <div class="form-check">
                            <input class="form-check-input producto-checkbox" type="checkbox" value="${producto.idProducto}" 
                                   data-nombre="${producto.nombreProducto}" data-precio="${producto.precio}" 
                                   id="productoAdicional${producto.idProducto}">
                            <label class="form-check-label" for="productoAdicional${producto.idProducto}">
                                ${producto.nombreProducto}
                            </label>
                        </div>
                        <span class="text-muted small">$${producto.precio}</span>
                    `;
                    productosAdicionalesContainer.appendChild(productoDiv);
                });
            } else {
                productosAdicionalesContainer.innerHTML = '<span class="text-muted">No hay productos adicionales disponibles.</span>';
            }

            // Cargar extras (decoradores con id_producto igual al producto seleccionado)
            const extrasContainer = document.getElementById('extrasContainer');
            extrasContainer.innerHTML = '';
            const decoradores = window.decoradores || [];
            const extras = decoradores.filter(d => d.idProducto == productoSeleccionado.idProducto);
            if (extras.length > 0) {
                extras.forEach(extra => {
                    const extraDiv = document.createElement('div');
                    extraDiv.className = "d-flex align-items-center justify-content-between mb-1";
                    extraDiv.innerHTML = `
                        <div class="form-check">
                            <input class="form-check-input extra-checkbox" type="checkbox" value="${extra.idDecorador}" data-nombre="${extra.nombre}" data-precio="${extra.precioExtra}" id="extra${extra.idDecorador}">
                            <label class="form-check-label" for="extra${extra.idDecorador}">
                                ${extra.nombre}
                            </label>
                        </div>
                        <span class="text-muted small">$${extra.precioExtra}</span>
                    `;
                    extrasContainer.appendChild(extraDiv);
                });
            } else {
                extrasContainer.innerHTML = '<span class="text-muted">No hay extras disponibles para este producto.</span>';
            }

            // Mostrar el modal
            var modal = new bootstrap.Modal(document.getElementById('modalAgregarCarrito'));
            modal.show();
        } else {
            var loginModal = new bootstrap.Modal(document.getElementById('loginModal'));
            loginModal.show();
        }
    });
}

// Evento para el bot√≥n de confirmar a√±adir al carrito - ACTUALIZAR ESTO TAMBI√âN
document.getElementById('btnConfirmarAgregarCarrito').addEventListener('click', async function() {
    if (productoSeleccionado) {
        const productosSeleccionados = [];
        document.querySelectorAll('#productosAdicionalesContainer .producto-checkbox:checked').forEach(cb => {
            productosSeleccionados.push(Number(cb.value));
        });
        const extrasSeleccionados = [];
        document.querySelectorAll('#extrasContainer .extra-checkbox:checked').forEach(cb => {
            extrasSeleccionados.push(Number(cb.value));
        });

        const item = {
            idProducto: productoSeleccionado.idProducto,
            nombre: productoSeleccionado.nombre,
            precioBase: parseFloat(productoSeleccionado.precio),
            cantidad: cantidadPedido,
            agrandado: agrandarCombo,
            productosAdicionales: productosSeleccionados,
            extras: extrasSeleccionados
        };

        await fetch('/carrito/agregar', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(item)
        });

        var modal = bootstrap.Modal.getInstance(document.getElementById('modalAgregarCarrito'));
        modal.hide();

        // Mostrar toast personalizado
        mostrarToastCarrito();
    }
});

}
